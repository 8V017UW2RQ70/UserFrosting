  8  * Controller class for /config/* URLs.  Handles admin-related activities, including site settings, etc

 13  */
 14 class AdminController extends \UserFrosting\BaseController {
 15 
 16     /**
 17      * Create a new AdminController object.
 18      *
 19      * @param UserFrosting $app The main UserFrosting app.
 20      */
 21     public function __construct($app){
 22         $this->_app = $app;
 23     }
 24 
 25     /**
 26      * Renders the site settings page.
 27      *
 28      * This page provides an interface for modifying site settings, especially those handled by the SiteSettings class.
 29      * It also shows some basic configuration information for the site, along with a nicely formatted readout of the PHP error log.
 30      * This page requires authentication (and should generally be limited to the root user).
 31      * Request type: GET
 32      */
 33     public function pageSiteSettings(){
 34         // Access-controlled page
 35         if (!$this->_app->user->checkAccess('uri_site_settings')){
 36             $this->_app->notFound();
 37         }
 38         
 39         // Hook for core and plugins to register their settings
 40         $this->_app->applyHook("settings.register");
 41         
 42         $this->_app->render('config/site-settings.twig', [
 43             'settings' => $this->_app->site->getRegisteredSettings(),
 44             'info'     => $this->_app->site->getSystemInfo(),
 45             'error_log'=> $this->_app->site->getLog(50)
 46         ]);    
 47     }
 48         
 49     /**
 50      * Processes a request to update the site settings.
 51      *
 52      * Processes the request from the site settings form, checking that:
 53      * 1. The setting name has been registered with the SiteSettings object.
 54      * This route requires authentication.
 55      * Request type: POST
 56      * @todo validate setting syntax
 57      */       
 58     public function siteSettings(){
 59         // Get the alert message stream
 60         $ms = $this->_app->alerts;
 61         
 62         $post = $this->_app->request->post();
 63         
 64         // Remove CSRF token
 65         if (isset($post['csrf_token']))
 66             unset($post['csrf_token']);
 67             
 68         // Access-controlled page
 69         if (!$this->_app->user->checkAccess('update_site_settings')){
 70             $ms->addMessageTranslated("danger", "ACCESS_DENIED");
 71             $this->_app->halt(403);
 72         }
 73         
 74         // Hook for core and plugins to register their settings
 75         $this->_app->applyHook("settings.register");
 76         
 77         // Get registered settings
 78         $registered_settings = $this->_app->site->getRegisteredSettings();
 79         
 80         // Ok, check that all posted settings are registered
 81         foreach ($post as $plugin => $settings){
 82             if (!isset($registered_settings[$plugin])){
 83                 $ms->addMessageTranslated("danger", "CONFIG_PLUGIN_INVALID", ["plugin" => $plugin]);
 84                 $this->_app->halt(400);
 85             }
 86             foreach ($settings as $name => $value){
 87                 if (!isset($registered_settings[$plugin][$name])){
 88                     $ms->addMessageTranslated("danger", "CONFIG_SETTING_INVALID", ["plugin" => $plugin, "name" => $name]);
 89                     $this->_app->halt(400);
 90                 }
 91             }
 92         }
 93         
 94         // TODO: validate setting syntax
 95         
 96         // If validation passed, then update
 97         foreach ($post as $plugin => $settings){
 98             foreach ($settings as $name => $value){
 99                 $this->_app->site->set($plugin, $name, $value);
100             }
101         }
102         $this->_app->site->store();
103     }

105 }
