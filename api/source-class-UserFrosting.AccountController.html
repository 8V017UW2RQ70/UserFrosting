189     /**
190      * Processes an account login request.
191      *
192      * Processes the request from the form on the login page, checking that:
193      * 1. The user is not already logged in.
194      * 2. Email login is enabled, if an email address was used.
195      * 3. The user account exists.
196      * 4. The user account is enabled and active.
197      * 5. The user entered a valid username/email and password.
198      * This route, by definition, is "public access".
199      * Request type: POST     
200      */        
201     public function login(){
202         // Load the request schema
203         $requestSchema = new \Fortress\RequestSchema($this->_app->config('schema.path') . "/forms/login.json");
204         
205         // Get the alert message stream
206         $ms = $this->_app->alerts; 
207         
208         // Forward the user to their default page if he/she is already logged in
209         if(!$this->_app->user->isGuest()) {
210             $ms->addMessageTranslated("warning", "LOGIN_ALREADY_COMPLETE");
211             $this->_app->halt(200);
212         }
213         
214         // Set up Fortress to process the request
215         $rf = new \Fortress\HTTPRequestFortress($ms, $requestSchema, $this->_app->request->post());
216         
217         // Sanitize data
218         $rf->sanitize();
219                 
220         // Validate, and halt on validation errors.
221         if (!$rf->validate(true)) {
222             $this->_app->halt(400);
223         }
224         
225         // Get the filtered data
226         $data = $rf->data();
227         
228         // Determine whether we are trying to log in with an email address or a username
229         $isEmail = filter_var($data['user_name'], FILTER_VALIDATE_EMAIL);
230         
231         // If it's an email address, but email login is not enabled, raise an error.
232         if ($isEmail && !$this->_app->site->email_login){
233             $ms->addMessageTranslated("danger", "ACCOUNT_USER_OR_PASS_INVALID");
234             $this->_app->halt(403);
235         }
236         
237         // Load user by email address
238         if($isEmail){
239             $user = User::where('email', $data['user_name'])->first();
240             if (!$user){
241                 $ms->addMessageTranslated("danger", "ACCOUNT_USER_OR_PASS_INVALID");
242                 $this->_app->halt(403);         
243             }
244         // Load user by user name    
245         } else {
246             $user = User::where('user_name', $data['user_name'])->first();
247             if (!$user) {
248                 $ms->addMessageTranslated("danger", "ACCOUNT_USER_OR_PASS_INVALID");
249                 $this->_app->halt(403);
250             }
251         }
252         
253         // Check that the user has a password set (so, rule out newly created accounts without a password)
254         if (!$user->password) {
255             $ms->addMessageTranslated("danger", "ACCOUNT_USER_OR_PASS_INVALID");
256             $this->_app->halt(403);
257         }
258         
259         // Check that the user's account is enabled
260         if ($user->flag_enabled == 0){
261             $ms->addMessageTranslated("danger", "ACCOUNT_DISABLED");
262             $this->_app->halt(403);
263         }        
264         
265         // Check that the user's account is activated
266         if ($user->flag_verified == 0) {
267             $ms->addMessageTranslated("danger", "ACCOUNT_INACTIVE");
268             $this->_app->halt(403);
269         }
270         
271         // Here is my password.  May I please assume the identify of this user now?
272         if ($user->verifyPassword($data['password']))  {
273             $this->_app->login($user, !empty($data['rememberme']));       
274             $ms->addMessageTranslated("success", "ACCOUNT_WELCOME", $this->_app->user->export());
275         } else {
276             //Again, we know the password is at fault here, but lets not give away the combination in case of someone bruteforcing
277             $ms->addMessageTranslated("danger", "ACCOUNT_USER_OR_PASS_INVALID");
278             $this->_app->halt(403);
279         }
280         
281     }

295     /**
296      * Processes an new account registration request.
297      *
298      * Processes the request from the form on the registration page, checking that:
299      * 1. The honeypot was not modified;
300      * 2. The master account has already been created (during installation);
301      * 3. Account registration is enabled;
302      * 4. The user is not already logged in;
303      * 5. Valid information was entered;
304      * 6. The captcha, if enabled, is correct;
305      * 7. The username and email are not already taken.
306      * Automatically sends an activation link upon success, if account activation is enabled.
307      * This route is "public access".
308      * Request type: POST     
309      */      
310     public function register(){
311         // POST: user_name, display_name, email, title, password, passwordc, captcha, spiderbro, csrf_token
312         $post = $this->_app->request->post();
313         
314         // Get the alert message stream
315         $ms = $this->_app->alerts; 
316         
317         // Check the honeypot. 'spiderbro' is not a real field, it is hidden on the main page and must be submitted with its default value for this to be processed.
318         if (!$post['spiderbro'] || $post['spiderbro'] != "http://"){
319             error_log("Possible spam received:" . print_r($this->_app->request->post(), true));
320             $ms->addMessage("danger", "Aww hellllls no!");
321             $this->_app->halt(500);     // Don't let on about why the request failed ;-)
322         }  
323                
324         // Load the request schema
325         $requestSchema = new \Fortress\RequestSchema($this->_app->config('schema.path') . "/forms/register.json");
326                    
327         // Set up Fortress to process the request
328         $rf = new \Fortress\HTTPRequestFortress($ms, $requestSchema, $post);        
329 
330         // Security measure: do not allow registering new users until the master account has been created.        
331         if (!User::find($this->_app->config('user_id_master'))){
332             $ms->addMessageTranslated("danger", "MASTER_ACCOUNT_NOT_EXISTS");
333             $this->_app->halt(403);
334         }
335           
336         // Check if registration is currently enabled
337         if (!$this->_app->site->can_register){
338             $ms->addMessageTranslated("danger", "ACCOUNT_REGISTRATION_DISABLED");
339             $this->_app->halt(403);
340         }
341           
342         // Prevent the user from registering if he/she is already logged in
343         if(!$this->_app->user->isGuest()) {
344             $ms->addMessageTranslated("danger", "ACCOUNT_REGISTRATION_LOGOUT");
345             $this->_app->halt(200);
346         }
347                 
348         // Sanitize data
349         $rf->sanitize();
350                 
351         // Validate, and halt on validation errors.
352         $error = !$rf->validate(true);
353         
354         // Get the filtered data
355         $data = $rf->data();        
356         
357         // Check captcha, if required
358         if ($this->_app->site->enable_captcha == "1"){
359             if (!$data['captcha'] || md5($data['captcha']) != $_SESSION['userfrosting']['captcha']){
360                 $ms->addMessageTranslated("danger", "CAPTCHA_FAIL");
361                 $error = true;
362             }
363         }
364         
365         // Remove captcha, password confirmation from object data
366         $rf->removeFields(['captcha', 'passwordc']);
367         
368         // Perform desired data transformations.  Is this a feature we could add to Fortress?
369         $data['display_name'] = trim($data['display_name']);
370         $data['locale'] = $this->_app->site->default_locale;
371         
372         if ($this->_app->site->require_activation)
373             $data['flag_verified'] = 0;
374         else
375             $data['flag_verified'] = 1;
376         
377         // Check if username or email already exists
378         if (User::where('user_name', $data['user_name'])->first()){
379             $ms->addMessageTranslated("danger", "ACCOUNT_USERNAME_IN_USE", $data);
380             $error = true;
381         }
382 
383         if (User::where('email', $data['email'])->first()){
384             $ms->addMessageTranslated("danger", "ACCOUNT_EMAIL_IN_USE", $data);
385             $error = true;
386         }
387         
388         // Halt on any validation errors
389         if ($error) {
390             $this->_app->halt(400);
391         }
392     
393         // Get default primary group (is_default = GROUP_DEFAULT_PRIMARY)
394         $primaryGroup = Group::where('is_default', GROUP_DEFAULT_PRIMARY)->first();
395         
396         // Check that a default primary group is actually set
397         if (!$primaryGroup){
398             $ms->addMessageTranslated("danger", "ACCOUNT_REGISTRATION_BROKEN");
399             error_log("Account registration is not working because a default primary group has not been set.");
400             $this->_app->halt(500);
401         }
402         
403         $data['primary_group_id'] = $primaryGroup->id;
404         // Set default title for new users
405         $data['title'] = $primaryGroup->new_user_title;
406         // Hash password
407         $data['password'] = Authentication::hashPassword($data['password']);
408         
409         // Create the user
410         $user = new User($data);
411 
412         // Add user to default groups, including default primary group
413         $defaultGroups = Group::where('is_default', GROUP_DEFAULT)->get();
414         $user->addGroup($primaryGroup->id);
415         foreach ($defaultGroups as $group)
416             $user->addGroup($group->id);    
417         
418         // Create sign-up event
419         $user->newEventSignUp();
420         
421         // Store new user to database
422         $user->save();
423         
424         if ($this->_app->site->require_activation) {
425             // Create verification request event
426             $user->newEventVerificationRequest();
427             $user->save();      // Re-save with verification event      
428             
429             // Create and send verification email
430             $twig = $this->_app->view()->getEnvironment();
431             $template = $twig->loadTemplate("mail/activate-new.twig");        
432             $notification = new Notification($template);
433             $notification->fromWebsite();      // Automatically sets sender and reply-to
434             $notification->addEmailRecipient($user->email, $user->display_name, [
435                 "user" => $user
436             ]);
437             
438             try {
439                 $notification->send();
440             } catch (\phpmailerException $e){
441                 $ms->addMessageTranslated("danger", "MAIL_ERROR");
442                 error_log('Mailer Error: ' . $e->errorMessage());
443                 $this->_app->halt(500);
444             }
445             
446             $ms->addMessageTranslated("success", "ACCOUNT_REGISTRATION_COMPLETE_TYPE2");
447         } else
448             // No activation required
449             $ms->addMessageTranslated("success", "ACCOUNT_REGISTRATION_COMPLETE_TYPE1");
450     }

562     /**
563      * Processes a request to reset a user's password, or set the password for a new user.
564      *
565      * Processes the request from the password create/reset form, which should have the secret token embedded in it, checking that:
566      * 1. The provided secret token is associated with an existing user account;
567      * 2. The user has a lost password request in progress;
568      * 3. The token has not expired;
569      * 4. The submitted data (new password) is valid.
570      * This route is "public access".
571      * Request type: POST     
572      */       
573     public function setPassword($flag_new_user = false){
574         $data = $this->_app->request->post();
575         
576         // Load the request schema
577         $requestSchema = new \Fortress\RequestSchema($this->_app->config('schema.path') . "/forms/set-password.json");
578         
579         // Get the alert message stream
580         $ms = $this->_app->alerts; 
581         
582         // Set up Fortress to validate the request
583         $rf = new \Fortress\HTTPRequestFortress($ms, $requestSchema, $data);
584         
585         // Validate
586         if (!$rf->validate()) {
587             $this->_app->halt(400);
588         }
589         
590         // Fetch the user, by looking up the submitted secret token
591         $user = User::where('secret_token', $data['secret_token'])->first();
592         
593         // If no user exists for this token, just say the token is invalid.
594         if (!$user){
595             $ms->addMessageTranslated("danger", "FORGOTPASS_INVALID_TOKEN");
596             $this->_app->halt(400);
597         }
598         
599         // Get the most recent password reset request time
600         $last_password_reset_time = $user->lastEventTime('password_reset_request');
601

803         
804         // Get the alert message stream
805         $ms = $this->_app->alerts; 
806         
807         // Access control for entire page
808         if (!$this->_app->user->checkAccess('uri_account_settings')){
809             $ms->addMessageTranslated("danger", "ACCESS_DENIED");
810             $this->_app->halt(403);
811         }
812         
813         $data = $this->_app->request->post();
814         
815         // Remove csrf_token
816         unset($data['csrf_token']);
817         
818         // Check current password
819         if (!isset($data['passwordcheck']) || !$this->_app->user->verifyPassword($data['passwordcheck'])){
820             $ms->addMessageTranslated("danger", "ACCOUNT_PASSWORD_INVALID");
821             $this->_app->halt(403);
822         }        
823         
824         // Validate new email, if specified
825         if (isset($data['email']) && $data['email'] != $this->_app->user->email){
826             // Check authorization
827             if (!$this->_app->user->checkAccess('update_account_setting', ['user' => $this->_app->user, 'property' => 'email'])){
828                 $ms->addMessageTranslated("danger", "ACCESS_DENIED");
829                 $this->_app->halt(403);
830             }
831             // Check if address is in use
832             if (User::where('email', $data['email'])->first()){
833                 $ms->addMessageTranslated("danger", "ACCOUNT_EMAIL_IN_USE", $data);
834                 $this->_app->halt(400);
835             }            
836         } else {
837             $data['email'] = $this->_app->user->email;
838         }
839             
840         // Validate locale, if specified
841         if (isset($data['locale']) && $data['locale'] != $this->_app->user->locale){
842             // Check authorization
843             if (!$this->_app->user->checkAccess('update_account_setting', ['user' => $this->_app->user, 'property' => 'locale'])){
844                 $ms->addMessageTranslated("danger", "ACCESS_DENIED");
845                 $this->_app->halt(403);
846             }
847             // Validate locale
848             if (!in_array($data['locale'], $this->_app->site->getLocales())){
849                 $ms->addMessageTranslated("danger", "ACCOUNT_SPECIFY_LOCALE");
850                 $this->_app->halt(400);
851             }

853             $data['locale'] = $this->_app->user->locale;

868         if (isset($data['password']) && !empty($data['password'])){
869             // Check authorization

882         
883         // Validate

885             $this->_app->halt(400);
886         }    
887

894         unset($data['passwordcheck']);

899         }

901         $this->_app->user->store();
902         
903         $ms->addMessageTranslated("success", "ACCOUNT_SETTINGS_UPDATED");
