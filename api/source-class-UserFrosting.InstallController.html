 16     /**
 17      * Renders the initial page that comes up when you first install UserFrosting.
 18      *
 19      * This page performs the following steps:
 20      * 1. Check that the current version of PHP is adequate.
 21      * 2. Check that PDO is installed and enabled.
 22      * 3. Check that we can connect to the database, as configured in `config-userfrosting.php`.
 23      * 4. Check that the database tables have not already been created.
 24      * 5. If all of these checks are passed, set up the initial tables by calling `Database::install()`.
 25      * This page is "public access".
 26      * Request type: GET
 27      * @see MySqlDatabase::install()
 28      */    
 29     public function pageSetupDB(){
 30         $messages = [];
 31         // 1. Check PHP version
 32         
 33         error_log("Checking php version");
 34         // PHP_VERSION_ID is available as of PHP 5.2.7, if our version is lower than that, then emulate it
 35         if (!defined('PHP_VERSION_ID')) {
 36             $version = explode('.', PHP_VERSION);
 37             define('PHP_VERSION_ID', ($version[0] * 10000 + $version[1] * 100 + $version[2]));
 38         }
 39         if (PHP_VERSION_ID < 50400){
 40             $messages[] = [
 41                 "title" => "You need to upgrade your PHP installation.",
 42                 "message" => "I'm sorry, UserFrosting relies on numerous features of PHP that are only available in PHP 5.4 or later.  Please upgrade your version of PHP, or contact your web hosting service and ask them to upgrade it for you."
 43             ];
 44         }
 45         
 46         // 2. Check that PDO is installed and enabled
 47         if (!class_exists('PDO')){
 48             $messages[] = [
 49                 "title" => "PDO is not installed.",
 50                 "message" => "I'm sorry, you must have PDO installed and enabled in order for UserFrosting to access the database.  If you don't know what PDO is, please see <a href='http://php.net/manual/en/book.pdo.php'>http://php.net/manual/en/book.pdo.php</a>.  You must also have MySQL version 4.1 or higher installed, since UserFrosting relies on native prepared statements."
 51             ];    
 52         }
 53         
 54         error_log("Checking db connection");
 55         // 3. Check database connection
 56         if (!Database::testConnection()){
 57             $messages[] = [
 58                 "title" => "We couldn't connect to your database.",
 59                 "message" => "Make sure that your database is properly configured in <code>config-userfrosting.php</code>, and that you have selected the correct configuration mode ('dev' or 'production').  Also, make sure that your database user has the proper privileges to connect to the database."
 60             ]; 
 61         } 
 62         
 63         error_log("Checking any current tables");
 64         $tables = Database::getCreatedTables();
 65         if (count($tables) > 0){
 66             $messages[] = [
 67                 "title" => "One or more tables already exist.",
 68                 "message" => "The following tables already exist in the database: <strong>" . implode(", ", $tables) . "</strong>.  Do you already have another installation of UserFrosting in this database?  Please either create a new database (recommended), or change the table prefix in <code>config-userfrosting.php</code> if you cannot create a new database."
 69             ]; 
 70         }
 71         error_log("Done with checks");
 72         if (count($messages) > 0){
 73             $this->_app->render('install/install-errors.twig', [
 74                 "messages" => $messages
 75             ]);
 76         } else {
 77         error_log("Installing");
 78             // Create tables
 79             Database::install();
 80             
 81             $messages[] = [
 82                 "title" => "<i class='fa fa-lock'></i> PDO is installed.",
 83                 "message" => "No need to worry about any pesky SQL injection attacks!",
 84                 "class" => "success"
 85             ];
 86             
 87             $messages[] = [
 88                 "title" => "<i class='fa fa-database'></i> Database connection",
 89                 "message" => "Hooray!  We were able to connect to your database and create the core tables for UserFrosting.",
 90                 "class" => "success"
 91             ];
 92             if (PHP_VERSION_ID >= 50400 && PHP_VERSION_ID < 50500){
 93                $messages[] = [
 94                     "title" => "<i class='fa fa-warning'></i> PHP version",
 95                     "message" => "You currently have version " . PHP_VERSION . " of PHP installed.  We recommend version 5.5 or later.  UserFrosting can still be installed, but we highly recommend you upgrade soon.", 
 96                     "class" => "warning"
 97                 ];
 98             } else {
 99                 $messages[] = [
100                     "title" => "<i class='fa fa-check'></i> PHP version",
101                     "message" => "You currently have version " . PHP_VERSION . " of PHP installed.  Good job!", 
102                     "class" => "success"
103                 ]; 
104             }            
105             
106             // Check for GD library (required for Captcha)
107             if (!(extension_loaded('gd') && function_exists('gd_info'))) {
108                 $messages[] = [
109                     "title" => "<i class='fa fa-warning'></i> GD library not installed",
110                     "message" => "We could not confirm that the <code>GD</code> library is installed and enabled.  GD is an image processing library that UserFrosting uses to generate captcha codes for user account registration.  If you don't need captcha, you can disable it in Site Settings and ignore this message.", 
111                     "class" => "warning"
112                 ];
113             }
114             
115             $this->_app->render('install/install-ready.twig', [
116                 "messages" => $messages
117             ]);        
118         }
119     }

122      * Renders the page for creating the master account.
123      *
124      * This page performs the following steps:
125      * 1. Check that the master account doesn't already exist.  If it does, redirect to the home page.
126      * 2. This page features a "configuration token" as a security feature, to prevent malicious agents from intercepting
127      * an in-progress installation and setting themselves as the master account.  This requires the developer to look
128      * in the configuration table of the database.
129      * Request type: GET
130      */
131     public function pageSetupMasterAccount(){
132         
133         // Get the alert message stream
134         $ms = $this->_app->alerts;
135         
136         // Do not allow registering a master account if one has already been created     
137         if (UserLoader::exists($this->_app->config('user_id_master'))){
138             $ms->addMessageTranslated("danger", "MASTER_ACCOUNT_EXISTS");
139             $this->_app->redirect($this->_app->urlFor('uri_home'));
140         }
141         
142         $schema = new \Fortress\RequestSchema($this->_app->config('schema.path') . "/forms/register.json");
143         $this->_app->jsValidator->setSchema($schema);  
144         
145         $this->_app->render('install/install-master.twig', [
146             'validators' => $this->_app->jsValidator->rules(),
147             'table_config' => Database::getSchemaTable('configuration')->name
148         ]);

151     /**
152      * Processes a request to create the master account.
153      *
154      * Processes the request from the master account creation form, checking that:
155      * 1. The honeypot has not been changed;
156      * 2. The master account does not already exist;
157      * 3. The correct configuration token was submitted;
158      * 3. The submitted data is valid.
159      * This route is "public access" (until the master account has been created, that is)
160      * Request type: POST     
161      */        
162     public function setupMasterAccount(){
163         $post = $this->_app->request->post();
164         
165         // Get the alert message stream
166         $ms = $this->_app->alerts;
167         
168         // Check the honeypot. 'spiderbro' is not a real field, it is hidden on the main page and must be submitted with its default value for this to be processed.
169         if (!$post['spiderbro'] || $post['spiderbro'] != "http://"){
170             error_log("Possible spam received:" . print_r($this->_app->request->post(), true));
171             $ms->addMessage("danger", "Aww hellllls no!");
172             $this->_app->halt(500);     // Don't let on about why the request failed ;-)
173         }  
174         
175         // Do not allow registering a master account if one has already been created     
176         if (UserLoader::exists($this->_app->config('user_id_master'))){
177             $ms->addMessageTranslated("danger", "MASTER_ACCOUNT_EXISTS");
178             $this->_app->halt(403); 
179         }        
180         
181         // Check the configuration token
182         if ($post['root_account_config_token'] != $this->_app->site->root_account_config_token) {
183             $ms->addMessageTranslated("danger", "CONFIG_TOKEN_MISMATCH");
184             $this->_app->halt(403);
185         }
186           
187         // Load the request schema
188         $requestSchema = new \Fortress\RequestSchema($this->_app->config('schema.path') . "/forms/register.json");
189                    
190         // Set up Fortress to process the request
191         $rf = new \Fortress\HTTPRequestFortress($ms, $requestSchema, $post);        
192         
193         // Sanitize data
194         $rf->sanitize();
195                 
196         // Validate, and halt on validation errors.
197         $error = !$rf->validate(true);
198         
199         // Get the filtered data
200         $data = $rf->data();        
201         
202         // Remove configuration token, password confirmation from object data
203         $rf->removeFields(['root_account_config_token', 'passwordc']);
204         
205         // Perform desired data transformations.  Is this a feature we could add to Fortress?
206         $data['user_name'] = strtolower(trim($data['user_name']));
207         $data['display_name'] = trim($data['display_name']);
208         $data['email'] = strtolower(trim($data['email']));
209         $data['active'] = 1;
210         $data['locale'] = $this->_app->site->default_locale;
211                 
212         // Halt on any validation errors
213         if ($error) {
214             $this->_app->halt(400);
215         }
216     
217         // Get default primary group (is_default = GROUP_DEFAULT_PRIMARY)
218         $primaryGroup = GroupLoader::fetch(GROUP_DEFAULT_PRIMARY, "is_default");
219         $data['primary_group_id'] = $primaryGroup->id;
220         // Set default title for new users
221         $data['title'] = $primaryGroup->new_user_title;
222         // Hash password
223         $data['password'] = Authentication::hashPassword($data['password']);
224             
225         // Create the user
226         $user = new User($data, $this->_app->config('user_id_master'));
227 
228         // Add user to default groups, including default primary group
229         $defaultGroups = GroupLoader::fetchAll(GROUP_DEFAULT, "is_default");
230         $user->addGroup($primaryGroup->id);
231         foreach ($defaultGroups as $group_id => $group)
232             $user->addGroup($group_id);    
233         
234         // Add sign-up event
235         $user->newEventSignUp();
236         
237         // Store new user to database
238         $user->save();
239         
240         // No activation required
241         $ms->addMessageTranslated("success", "ACCOUNT_REGISTRATION_COMPLETE_TYPE1");
242         
243         // Update install status
244         $this->_app->site->install_status = "new";
245         $this->_app->site->root_account_config_token = "";
246         $this->_app->site->store();
247     }
