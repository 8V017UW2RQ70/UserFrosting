445     /**
446      * Checks whether or not this user has access for a particular authorization hook.
447      *
448      * Determine if this user has access to the given $hook under the given $params.
449      * @param string $hook The authorization hook to check for access.
450      * @param array $params[optional] An array of field names => values, specifying any additional data to provide the authorization module
451      * when determining whether or not this user has access.
452      * @return boolean True if the user has access, false otherwise.
453      */ 
454     public function checkAccess($hook, $params = []){
455         if ($this->isGuest()){   // TODO: do we sometimes want to allow access to protected resources for guests?  Should we model a "guest" group?
456             return false;
457         }
458     
459         // The master (root) account has access to everything.
460         if ($this->id == static::$app->config('user_id_master'))
461             return true;
462              
463         // Try to find an authorization rule for $hook that matches the currently logged-in user, or one of their groups.
464         $rule = UserAuth::fetchUserAuthHook($this->id, $hook);
465         
466         if (empty($rule))
467             $pass = false;
468         else {      
469             $ace = new AccessConditionExpression(static::$app); // TODO: should we have to pass the app in, or just make it available statically?
470             $pass = $ace->evaluateCondition($rule['conditions'], $params);
471         }
472         
473         // If no user-specific rule is passed, look for a group-level rule
474         if (!$pass){
475             $ace = new AccessConditionExpression(static::$app);
476             $groups = $this->getGroupIds();
477             foreach ($groups as $group_id){
478                 // Try to find an authorization rule for $hook that matches this group
479                 $rule = GroupAuth::fetchGroupAuthHook($group_id, $hook);
480                 if (!$rule)
481                     continue;
482                 $pass = $ace->evaluateCondition($rule['conditions'], $params);
483                 if ($pass)
484                     break;
485             }
486         }
487         return $pass;
488     }

490     /**
491      * Verify a plaintext password against the user's hashed password.
492      *
493      * @param string $password The plaintext password to verify.
494      * @return boolean True if the password matches, false otherwise.
495      */   
496     public function verifyPassword($password){
497         if (Authentication::getPasswordHashType($this->password) == "sha1"){
498             $salt = substr($this->password, 0, 25);     // Extract the salt from the hash
499             $hash_input = $salt . sha1($salt . $password);
500             if ($hash_input == $this->password){
501                 return true;
502             } else {
503                 return false;
504             }
505         }   
506         // Homegrown implementation (assuming that current install has been using a cost parameter of 12)
507         else if (Authentication::getPasswordHashType($this->password) == "homegrown"){
508             /*used for manual implementation of bcrypt*/
509             $cost = '12'; 
510             if (substr($this->password, 0, 60) == crypt($password, "$2y$".$cost."$".substr($this->password, 60))){
511                 return true;
512             } else {
513                 return false;
514             }
515         // Modern implementation
516         } else {
517             return password_verify($password, $this->password);
518         }    
519     }

521     /**
522      * Log this user in.  This basically updates the user's sign-in time, and updates any old password hashes.
523      *
524      * You assign this user's id to $_SESSION["userfrosting"]["user_id"] after calling login, so that it will persist in the session.
525      */
526     public function login(){    
527         // Add a sign in event (time is automatically set by database)
528         $this->newEventSignIn();
529         
530         // Update password if we had encountered an outdated hash
531         if (Authentication::getPasswordHashType($this->password) != "modern"){
532             // Hash the user's password and update
533             $password_hash = Authentication::getPasswordHashType($password);
534             if ($password_hash === null){
535                 error_log("Notice: outdated password hash could not be updated because the new hashing algorithm is not supported.  Are you running PHP >= 5.3.7?");
536             } else {
537                 $this->password = $password_hash;
538                 error_log("Notice: outdated password hash has been automatically updated to modern hashing.");
539             }
540         }
541         
542         // Save changes
543         $this->save();
544         
545         return $this;
546     }

549      * Generate an activation token for a user.
550      *
551      * This generates a token to use for activating a new account, resetting a lost password, etc.
552      * @param string $gen specify an existing token that, if we happen to generate the same value, we should regenerate on.
553      * @return string
554      */
555     public static function generateActivationToken($gen = null) {
556         do {
557             $gen = md5(uniqid(mt_rand(), false));
558         } while(User::where('secret_token', $gen)->first());
559         return $gen;
