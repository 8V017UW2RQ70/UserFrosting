16     /**
17      * Get a list of the most recent event of a given type (e.g., "sign_in") for each user.
18      *
19      * On success, this will set the event times for the specified event type as attributes of the User models.
20      * Unfortunately, we can't store $recent_event_times directly in the collection, because it won't get copied in other operations.
21      * So, it must be returned by this method instead.
22      * @see https://github.com/laravel/framework/issues/10695
23      * @param string $type The type of event, matching the `event_type` field in the user_event table.
24      * @param string $field_name optional The attribute name to use for this event in the User model.  If not specified, defaults to last_{$type}_time.
25      * @return array An array of datetime strings, keyed by the `id` of each User.
26      */
27     public function getRecentEvents($type, $field_name = null) {        
28         if (!$field_name)
29             $field_name = "last_" . $type . "_time";
30         
31         $recentEventsQuery = UserEvent::mostRecentEventsByType($type);
32         $recent_events = $recentEventsQuery->get();
33         
34         $recent_event_times = [];
35         
36         // extract sign-in times
37         foreach($recent_events as $event){
38             $recent_event_times[$event['user_id']] = $event['occurred_at'];
39         }        
40         
41         // Merge in recent event times, and set any missing values
42         foreach ($this as $user){
43             if (isset($recent_event_times[$user->id]))
44                 $user->{$field_name} = $recent_event_times[$user->id];
45             else {
46                 $user->{$field_name} = 0;   // Should be an integer, so it can be properly parsed on client-side
47                 $recent_event_times[$user->id] = 0;
48             }
49         }
50         
51         return $recent_event_times;
52     }

54     /**
55      * Filter this collection based on a recent event time.
56      *
57      * This method allows you to filter Users by recent event times.  For example, you may filter users by their last sign-in time,
58      * last sign-up time, last activation request time, etc.
59      * @param string $type The type of event, matching the `event_type` field in the user_event table.
60      * @param array @event_times An array of datetimes, indexed by the Users' `id`s.  This is typically the output of getRecentEvents().
61      * @param string $value The value to search for.  This filter does a simple stripos for searching.
62      * @param string $format_zero optional The date format to use when searching zero-valued dates.
63      * @param string $format optional The date format to use when searching non-zero dates.  By default lets you search weekday, month, day, year, etc.
64      * @return UserCollection The modified UserCollection.
65      */
66     public function filterRecentEventTime($type, $event_times, $value, $format_zero = "Brand New!", $format = "l F j, Y g:i a"){
67         $result = $this->filter(function ($item) use ($value, $event_times, $format, $format_zero){
68             // Get the recent event time from the stored array.  If we try to get it from the object itself, it will requery!
69             $item_id = $item->id;
70             if (isset($event_times[$item_id])) {
71                 $stamp = strtotime($event_times[$item_id]);
72                 $last_event_time = (($stamp != 0) ? date($format, $stamp) : $format_zero);
73                 return (stripos($last_event_time, $value) !== false);
74             } else {
75                 return false;
76             }
77         });
78         return $result;
79     }

90         $result = $this->filter(function ($item) use ($name, $value){
91             return (stripos($item->{$name}, $value) !== false);
92         });
